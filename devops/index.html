edit your nginx.conf

    location /<url> {
      alias /usr/share/nginx/html/<link_name>
      autoindex on;
      autoindex_exact_size off;
      autoindex_localtime on;
      auth_basic "Restricted";                #For Basic Auth
      auth_basic_user_file <your .htpasswd>;  #For Basic Auth
    }
