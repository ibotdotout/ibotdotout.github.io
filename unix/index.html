  1. Parallelize Bash Script
#!/bin/bash

nproc=$(($(nproc)-1))  # number of core cpu - 1

# waitproc - wait until have core slots
waitproc () {
  jobcnt=($(jobs -p))
  until [ ${#jobcnt[@]} -lt $nproc ]; do
    sleep 0.05
    jobcnt=($(job -p))
  done
